\usepackage[newdimens]{labels}
\LabelCols=5                % Number of columns of labels per page
\LabelRows=17               % Number of rows of labels per page

\RightPageMargin=0.77in     % page gutter sizes. The outer edges of
\TopPageMargin=0.24in       % the outer labels are the specified
\BottomPageMargin=0.24in    % distances from the edge of the paper.

\InterLabelColumn=0.12in    % Gap between columns of labels
\InterLabelRow=0.13in       % Gap between rows of labels

\LeftLabelBorder=0mm        % These four parameters give the extra
\RightLabelBorder=0mm       % space used around the text on each
\TopLabelBorder=0mm         % actual label.
\BottomLabelBorder=0mm      %

\LabelInfotrue
